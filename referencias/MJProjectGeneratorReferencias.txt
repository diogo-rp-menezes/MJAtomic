  // Erro de valida√ß√£o (ex: Zod)

    });
  }
  
  // Erro inesperado
  const response: ErrorResponse = {
    error: 'InternalServerError',
    message: 'Erro interno do servidor'
  };
  
  // Detalhes apenas em desenvolvimento
  if (process.env.NODE_ENV === 'development') {

  }

}
